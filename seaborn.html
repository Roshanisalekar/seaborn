Sample DataFrames (with Pandas)

Seaborn is designed to work perfectly with Pandas DataFrames. Copy this setup code to follow the examples.


import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Sample Data 1: Monthly Data
data_monthly = {
    'month': ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'],
    'sales': [22, 25, 30, 35, 42, 45, 50, 48, 43, 38, 32, 28],
    'ad_spend': [1.5, 1.8, 2.0, 2.5, 3.0, 3.2, 3.5, 3.3, 3.1, 2.7, 2.4, 2.0]
}
df_monthly = pd.DataFrame(data_monthly)

# Sample Data 2: Product Data (Tidy Format)
data_products = {
    'category': ['Electronics', 'Clothing', 'Groceries', 'Books', 'Home Goods', 
                 'Electronics', 'Clothing', 'Groceries', 'Books', 'Home Goods'],
    'sale_price': [150, 50, 15, 25, 75, 180, 65, 12, 30, 85],
    'store_location': ['A', 'A', 'A', 'A', 'A', 'B', 'B', 'B', 'B', 'B']
}
df_products = pd.DataFrame(data_products)

# Note: We still need 'matplotlib.pyplot as plt' for plt.show()
